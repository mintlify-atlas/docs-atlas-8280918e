---
title: 'List endpoints'
api: 'GET /api/v1/endpoints/'
---

Retrieve a list of all endpoints in your tenant with summary information.

## Authentication

Requires tenant authentication via SyftBox token.

## Response

Returns an array of endpoint objects with summary information.

<ResponseField name="id" type="string" required>
  Unique identifier for the endpoint.
</ResponseField>

<ResponseField name="name" type="string" required>
  Name of the endpoint.
</ResponseField>

<ResponseField name="slug" type="string" required>
  URL-friendly unique identifier.
</ResponseField>

<ResponseField name="summary" type="string" required>
  Brief description of the endpoint.
</ResponseField>

<ResponseField name="response_type" type="string" required>
  Type of response (raw/summary/both).
</ResponseField>

<ResponseField name="published" type="boolean" required>
  Whether the endpoint is published.
</ResponseField>

<ResponseField name="tags" type="string" required>
  Comma-separated tags.
</ResponseField>

<ResponseField name="created_at" type="string" required>
  ISO 8601 timestamp of when the endpoint was created.
</ResponseField>

<ResponseField name="model" type="object">
  Attached model information, if any.
  
  <Expandable title="model properties">
    <ResponseField name="id" type="string" required>
      Unique identifier for the model.
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Name of the model.
    </ResponseField>
    
    <ResponseField name="dtype" type="string" required>
      Model type name (e.g., "vllm", "openai").
    </ResponseField>
    
    <ResponseField name="configuration" type="object" required>
      Model configuration settings.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="dataset" type="object">
  Attached dataset information, if any.
  
  <Expandable title="dataset properties">
    <ResponseField name="id" type="string" required>
      Unique identifier for the dataset.
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Name of the dataset.
    </ResponseField>
    
    <ResponseField name="summary" type="string" required>
      Brief description of the dataset.
    </ResponseField>
    
    <ResponseField name="dtype" type="string" required>
      Dataset type name (e.g., "weaviate", "qdrant").
    </ResponseField>
    
    <ResponseField name="configuration" type="object" required>
      Dataset configuration settings.
    </ResponseField>
  </Expandable>
</ResponseField>
