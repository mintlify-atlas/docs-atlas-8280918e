---
title: Get dataset
description: Retrieve details of a specific dataset by name
---

Retrieve complete details of a dataset including configuration, provisioner state, and connected endpoints.

## Request

<ParamField path="GET" type="string">
  `/datasets/{name}`
</ParamField>

### Path parameters

<ParamField path="name" type="string" required>
  The unique name of the dataset to retrieve
</ParamField>

### Authentication

Requires authentication via tenant credentials.

## Response

<ResponseField name="id" type="string">
  Unique identifier (UUID) for the dataset
</ResponseField>

<ResponseField name="name" type="string">
  Dataset name
</ResponseField>

<ResponseField name="dtype" type="string">
  Dataset type name (e.g., `local_file`, `remote_weaviate`)
</ResponseField>

<ResponseField name="configuration" type="object">
  Full configuration for the dataset. Structure varies by dataset type.
</ResponseField>

<ResponseField name="summary" type="string">
  Dataset summary
</ResponseField>

<ResponseField name="tags" type="string">
  Comma-separated tags
</ResponseField>

<ResponseField name="provisioner_state" type="object" nullable>
  Detailed provisioner state information (for local dataset types)

  <Expandable title="properties">
    <ResponseField name="status" type="string">
      Provisioner status: `stopped`, `starting`, `running`, `stopping`, or `error`
    </ResponseField>

    <ResponseField name="state" type="object" nullable>
      Provisioner state including connection config (ports) and runtime info (container_id, etc.)
    </ResponseField>

    <ResponseField name="started_at" type="string" nullable>
      ISO 8601 timestamp when provisioner was started
    </ResponseField>

    <ResponseField name="stopped_at" type="string" nullable>
      ISO 8601 timestamp when provisioner was stopped
    </ResponseField>

    <ResponseField name="error" type="string" nullable>
      Error message if status is `error`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of creation
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of last update
</ResponseField>

<ResponseField name="connected_endpoints" type="array">
  List of endpoints connected to this dataset

  <Expandable title="items">
    <ResponseField name="id" type="string">
      Endpoint unique identifier
    </ResponseField>

    <ResponseField name="name" type="string">
      Endpoint name
    </ResponseField>

    <ResponseField name="slug" type="string">
      Endpoint URL slug
    </ResponseField>
  </Expandable>
</ResponseField>
